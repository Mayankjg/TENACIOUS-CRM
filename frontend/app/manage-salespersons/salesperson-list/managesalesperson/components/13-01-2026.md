"use client";

import { useRouter } from "next/navigation";
import React, { useEffect, useState } from "react";
import { Mail, Phone, Briefcase, Trash2, Key } from "lucide-react";

import NewPasswordModal from "./NewPasswordModal";
import ChangeEmailModal from "./ChangeEmailModal";

export default function SalespersonCard() {
  const router = useRouter();

  const [salespersons, setSalespersons] = useState([]);
  const [searchQuery, setSearchQuery] = useState("");

  const [isModalOpen, setIsModalOpen] = useState(false);
  const [selectedId, setSelectedId] = useState(null);

  const [isEmailModalOpen, setIsEmailModalOpen] = useState(false);
  const [selectedEmailId, setSelectedEmailId] = useState(null);

  const API_BASE = "https://crm-tenacious-techies-pro-1.onrender.com/api";

  // FETCH FROM BACKEND
  useEffect(() => {
    const fetchList = async () => {
      try {
        const res = await fetch(`${API_BASE}/salespersons/get-salespersons`);
        const data = await res.json();
        setSalespersons(data);
      } catch (error) {
        console.log(error);
      }
    };
    fetchList();
  }, []);

  // DELETE SP
  const deleteSP = async (id) => {
    if (!confirm("Delete this salesperson?")) return;

    try {
      await fetch(`${API_BASE}/salespersons/delete-salesperson/${id}`, {
        method: "DELETE",
      });

      setSalespersons((prev) => prev.filter((sp) => sp.id !== id));
    } catch (error) {
      console.log(error);
    }
  };

  const resolveImageSrc = (profileImage) => {
    if (!profileImage)
      return `${API_BASE.replace(/\/api$/, "")}/uploads/default-avatar.png`;

    if (/^https?:\/\//i.test(profileImage)) return profileImage;

    if (profileImage.startsWith("/uploads")) {
      return `${API_BASE.replace(/\/api$/, "")}${profileImage}`;
    }

    return profileImage;
  };

  const filtered = searchQuery
    ? salespersons.filter(
        (sp) =>
          (sp.username || "")
            .toLowerCase()
            .includes(searchQuery.toLowerCase()) ||
          (sp.email || "").toLowerCase().includes(searchQuery.toLowerCase())
      )
    : salespersons;

  return (
    <div className="bg-[#f9fafb] p-3 sm:p-5 min-h-[80vh] flex justify-center">
      {/* PASSWORD MODAL */}
      {isModalOpen && (
        <NewPasswordModal
          salespersonId={selectedId}
          onClose={() => setIsModalOpen(false)}
          onPasswordChange={async (id, newPass) => {
            try {
              await fetch(`${API_BASE}/salespersons/update-salesperson-password/${id}`, {
                method: "PUT",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ newPassword: newPass }),
              });
              alert("Password Updated!");
            } catch (e) {
              console.log(e);
            }
          }}
        />
      )}

      {/* EMAIL MODAL */}
      {isEmailModalOpen && (
        <ChangeEmailModal
          salespersonId={selectedEmailId}
          onClose={() => setIsEmailModalOpen(false)}
          onEmailChange={async (id, email) => {
            try {
              const res = await fetch(`${API_BASE}/salespersons/update-salesperson-email/${id}`, {
                method: "PUT",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ email }),
              });

              const updated = await res.json();

              setSalespersons((prev) =>
                prev.map((sp) =>
                  sp.id === id ? { ...sp, email: updated.email } : sp
                )
              );
            } catch (e) {
              console.log(e);
            }
          }}
        />
      )}

      <div className="bg-white w-full max-w-[1400px] border border-black text-black shadow-lg">
        {/* ✅ HEADER - COMPACT */}
        <div className="py-3 sm:py-4 bg-gradient-to-r from-gray-50 to-white">
          <div className="px-4 sm:px-6">
            <div className="flex flex-col sm:flex-row justify-between gap-3 sm:items-center">
              <h2 className="text-xl sm:text-2xl text-gray-700 font-light">
                Salesperson <strong className="font-bold">List</strong>
              </h2>

              <button
                onClick={() =>
                  router.push(
                    "/manage-salespersons/salesperson-list/managesalesperson/add"
                  )
                }
                className="bg-[#374151] hover:bg-[#1f2937] text-white text-sm px-4 py-1.5 rounded shadow-sm transition-all duration-200 hover:shadow-md"
              >
                + Add Sales Person
              </button>
            </div>
          </div>

          <hr className="border-gray-300 mt-3 sm:mt-4" />
        </div>

        {/* ✅ SEARCH BAR - COMPACT */}
        <div className="flex flex-col sm:flex-row justify-end px-4 sm:px-6 py-3 gap-2 bg-gray-50">
          <input
            type="text"
            placeholder="Search by name or email..."
            className="border border-gray-300 px-3 py-1.5 w-full sm:w-[260px] rounded text-sm outline-none focus:border-blue-400 focus:ring-1 focus:ring-blue-200 transition-all"
            value={searchQuery}
            onChange={(e) => setSearchQuery(e.target.value)}
          />

          <button className="bg-[#18a3dd] text-white text-sm px-4 py-1.5 rounded hover:bg-[#1095cc] shadow-sm transition-all duration-200 hover:shadow-md">
            Search
          </button>
        </div>

        {/* ✅ LIST - PERFECT SPACING */}
        {filtered.length === 0 ? (
          <p className="text-center py-12 text-gray-500 text-sm">
            No Salespersons Found
          </p>
        ) : (
          <div className="px-4 sm:px-6 py-4 space-y-3">
            {filtered.map((sp) => {
              const imageSrc = resolveImageSrc(sp.profileImage);

              return (
                <div
                  key={sp.id}
                  className="p-4 bg-white border border-gray-200 rounded-lg hover:border-gray-300 hover:shadow-md transition-all duration-200"
                >
                  {/* ✅ DESKTOP LAYOUT - 4 COLUMNS WITH PERFECT SPACING */}
                  <div className="hidden lg:grid lg:grid-cols-12 gap-6 items-start">
                    
                    {/* ✅ COLUMN 1: PROFILE (4 cols) - INCREASED FONT SIZE */}
                    <div className="col-span-4 flex gap-3">
                      <img
                        src={imageSrc || "/default-avatar.png"}
                        className="w-[65px] h-[85px] border border-gray-300 rounded object-cover flex-shrink-0 shadow-sm"
                        alt={sp.username || "salesperson"}
                      />

                      <div className="flex-1 min-w-0">
                        <h3 className="text-lg font-bold text-gray-800 mb-1">
                          {sp.username}
                        </h3>

                        <p className="text-sm text-gray-600 mb-2">
                          {sp.firstname} {sp.lastname}
                        </p>

                        <div className="flex items-start gap-2">
                          <Mail className="w-4 h-4 text-gray-400 flex-shrink-0 mt-0.5" />
                          <a
                            href={`mailto:${sp.email}`}
                            className="text-blue-600 text-sm hover:text-blue-700 hover:underline break-all leading-tight"
                          >
                            {sp.email}
                          </a>
                        </div>
                      </div>
                    </div>

                    {/* ✅ COLUMN 2: DESIGNATION & CONTACT (3 cols) - INCREASED FONT SIZE */}
                    <div className="col-span-3 flex flex-col gap-2.5 pl-2 border-l border-gray-200">
                      <div className="flex items-start gap-2">
                        <Briefcase className="w-4 h-4 text-gray-400 flex-shrink-0 mt-0.5" />
                        <div className="flex-1 min-w-0">
                          <span className="text-xs text-gray-500 block">Designation</span>
                          <strong className="text-sm text-gray-800 block mt-0.5 break-words">
                            {sp.designation}
                          </strong>
                        </div>
                      </div>

                      <div className="flex items-start gap-2">
                        <Phone className="w-4 h-4 text-gray-400 flex-shrink-0 mt-0.5" />
                        <div className="flex-1 min-w-0">
                          <span className="text-xs text-gray-500 block">Contact</span>
                          <strong className="text-sm text-gray-800 block mt-0.5 break-all">
                            {sp.contact}
                          </strong>
                        </div>
                      </div>
                    </div>

                    {/* ✅ COLUMN 3: ACTIONS (2.5 cols) - INCREASED FONT SIZE */}
                    <div className="col-span-2 flex flex-col gap-2 pl-2 border-l border-gray-200">
                      <button
                        onClick={() => deleteSP(sp.id)}
                        className="flex items-center gap-2 text-red-600 hover:text-red-700 text-sm font-medium transition-colors py-1"
                      >
                        <Trash2 className="w-4 h-4 flex-shrink-0" />
                        <span>Delete</span>
                      </button>

                      <button
                        onClick={() => {
                          setSelectedId(sp.id);
                          setIsModalOpen(true);
                        }}
                        className="flex items-center gap-2 text-blue-700 hover:text-blue-800 text-sm font-medium transition-colors py-1 whitespace-nowrap"
                      >
                        <Key className="w-4 h-4 flex-shrink-0" />
                        <span>Change Password</span>
                      </button>
                    </div>

                    {/* ✅ COLUMN 4: MAIN BUTTONS (2.5 cols) - WITH LEFT GAP */}
                    <div className="col-span-3 flex flex-col gap-2 pl-4 border-l border-gray-200">
                      <button className="bg-red-500 hover:bg-red-600 text-white text-xs font-medium px-2.5 py-1.5 rounded shadow-sm transition-all duration-200 hover:shadow-md w-fit">
                        View Leads
                      </button>

                      <button
                        onClick={() => {
                          setSelectedEmailId(sp.id);
                          setIsEmailModalOpen(true);
                        }}
                        className="bg-[#2b3342] hover:bg-[#1f2937] text-white text-xs font-medium px-2.5 py-1.5 rounded shadow-sm transition-all duration-200 hover:shadow-md w-fit"
                      >
                        Change Email ID
                      </button>
                    </div>
                  </div>

                  {/* ✅ TABLET & MOBILE LAYOUT - INCREASED FONT SIZES */}
                  <div className="lg:hidden space-y-4">
                    {/* PROFILE SECTION */}
                    <div className="flex gap-3 pb-3 border-b border-gray-200">
                      <img
                        src={imageSrc || "/default-avatar.png"}
                        className="w-[60px] h-[80px] border border-gray-300 rounded object-cover flex-shrink-0 shadow-sm"
                        alt={sp.username || "salesperson"}
                      />

                      <div className="flex-1 min-w-0">
                        <h3 className="text-lg font-bold text-gray-800 mb-1">
                          {sp.username}
                        </h3>

                        <p className="text-sm text-gray-600 mb-2">
                          {sp.firstname} {sp.lastname}
                        </p>

                        <div className="flex items-start gap-2">
                          <Mail className="w-4 h-4 text-gray-400 flex-shrink-0 mt-0.5" />
                          <a
                            href={`mailto:${sp.email}`}
                            className="text-blue-600 text-sm hover:underline break-all"
                          >
                            {sp.email}
                          </a>
                        </div>
                      </div>
                    </div>

                    {/* INFO SECTION */}
                    <div className="grid grid-cols-1 sm:grid-cols-2 gap-3 pb-3 border-b border-gray-200">
                      <div className="flex items-start gap-2">
                        <Briefcase className="w-4 h-4 text-gray-400 flex-shrink-0 mt-0.5" />
                        <div className="flex-1 min-w-0">
                          <span className="text-xs text-gray-500 block">Designation</span>
                          <strong className="text-sm text-gray-800 block mt-0.5 break-words">
                            {sp.designation}
                          </strong>
                        </div>
                      </div>

                      <div className="flex items-start gap-2">
                        <Phone className="w-4 h-4 text-gray-400 flex-shrink-0 mt-0.5" />
                        <div className="flex-1 min-w-0">
                          <span className="text-xs text-gray-500 block">Contact</span>
                          <strong className="text-sm text-gray-800 block mt-0.5 break-all">
                            {sp.contact}
                          </strong>
                        </div>
                      </div>
                    </div>

                    {/* ACTIONS SECTION - DECREASED BUTTON WIDTHS */}
                    <div className="flex flex-wrap gap-2">
                      <button
                        onClick={() => deleteSP(sp.id)}
                        className="flex items-center gap-1.5 text-red-600 hover:text-red-700 text-sm font-medium px-2.5 py-1.5 border border-red-300 rounded hover:bg-red-50 transition-colors"
                      >
                        <Trash2 className="w-3.5 h-3.5" />
                        Delete
                      </button>

                      <button
                        onClick={() => {
                          setSelectedId(sp.id);
                          setIsModalOpen(true);
                        }}
                        className="flex items-center gap-1.5 text-blue-700 hover:text-blue-800 text-sm font-medium px-2.5 py-1.5 border border-blue-300 rounded hover:bg-blue-50 transition-colors"
                      >
                        <Key className="w-3.5 h-3.5" />
                        Change Password
                      </button>

                      <button className="bg-red-500 hover:bg-red-600 text-white text-xs font-medium px-2.5 py-1.5 rounded shadow-sm transition-all w-fit">
                        View Leads
                      </button>

                      <button
                        onClick={() => {
                          setSelectedEmailId(sp.id);
                          setIsEmailModalOpen(true);
                        }}
                        className="bg-[#2b3342] hover:bg-[#1f2937] text-white text-xs font-medium px-2.5 py-1.5 rounded shadow-sm transition-all w-fit"
                      >
                        Change Email ID
                      </button>
                    </div>
                  </div>
                </div>
              );
            })}
          </div>
        )}
      </div>
    </div>
  );
}
